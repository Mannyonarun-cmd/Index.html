// ── Voice Configuration (Glenn's style) ──────────────────────
function configureVoice() {
  if (!('speechSynthesis' in window)) return;
  const voices = window.speechSynthesis.getVoices();
  if (!voices.length) return;
  
  const maleVoices = voices.filter(v => 
    /en/i.test(v.lang) && (
      /fred|daniel|alex|david|bruce|ralph|tom/i.test(v.name.toLowerCase()) ||
      v.name.toLowerCase().includes('male')
    )
  );
  
  if (maleVoices.length) {
    preferredVoice = maleVoices.find(v => /fred/i.test(v.name)) || maleVoices[0];
  } else {
    preferredVoice = voices.find(v => /en/i.test(v.lang)) || voices[0];
  }
  voiceConfigured = true;
}

// ── Speech synthesis (Glenn-style tuning) ─────────────────────
function speakText(text, onDone) {
  if (!('speechSynthesis' in window)) { if (onDone) onDone(); return; }
  window.speechSynthesis.cancel();
  isSpeaking = true;
  updateOrb('SPEAKING');
  
  const u = new SpeechSynthesisUtterance(text);
  u.rate = 0.82;
  u.pitch = 0.78;
  u.volume = 1.0;
  
  if (preferredVoice) {
    u.voice = preferredVoice;
  }
  
  u.onend = () => { 
    isSpeaking = false; 
    updateOrb('IDLE'); 
    if (onDone) onDone(); 
  };
  u.onerror = (e) => { 
    isSpeaking = false; 
    updateOrb('IDLE'); 
    if (onDone) onDone(); 
  };
  
  window.speechSynthesis.speak(u);
}

  detections.forEach((det, i) => {
    const b = det.boundingBox;
    const x = (b.originX/sw)*canvas.width, y = (b.originY/sh)*canvas.height;
    const w = (b.width/sw)*canvas.width,   h = (b.height/sh)*canvas.height;
    ctx.strokeStyle='#ff0040'; ctx.lineWidth=2.5;
    ctx.strokeRect(x,y,w,h);
    ctx.fillStyle='rgba(255,0,64,0.88)'; ctx.fillRect(x,y-22,134,19);
    ctx.fillStyle='#fff'; ctx.font='11px Share Tech Mono, monospace';
    ctx.fillText('TARGET '+(i+1)+' ACQUIRED',x+6,y-8);
  });
}

// ── Mic ───────────────────────────────────────────────────────
function startListening(e) {
  if (e) { e.preventDefault(); e.stopPropagation(); }
  if (isListening || isSpeaking) return;

  const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
  if (!SR) {
    document.getElementById('micStatus').textContent = 'speech not supported. use chrome or safari.';
    return;
  }

  recognition = new SR();
  recognition.continuous = false;
  recognition.interimResults = true;
  recognition.lang = 'en-US';
  recognition.maxAlternatives = 1;
  isListening = true;

  document.getElementById('micBtn').classList.add('recording');
  document.getElementById('micStatus').textContent = '🔴 listening...';
  document.getElementById('transcriptPill').style.display = 'none';
  updateOrb('LISTENING');

  let finalText = '';

  recognition.onresult = (event) => {
    let interim = ''; finalText = '';
    for (let i = event.resultIndex; i < event.results.length; i++) {
      const t = event.results[i][0].transcript;
      if (event.results[i].isFinal) finalText += t;
      else interim += t;
    }
    const show = finalText || interim;
    const pill = document.getElementById('transcriptPill');
    pill.style.display = 'block';
    pill.innerHTML = '<span class="transcript-pill">' + show + '</span>';
  };

  recognition.onend = async () => {
    isListening = false;
    document.getElementById('micBtn').classList.remove('recording');

    const pill = document.getElementById('transcriptPill');
    const inner = pill.querySelector('.transcript-pill');
    const spoken = (inner ? inner.textContent : finalText).trim();

    if (!spoken) {
      document.getElementById('micStatus').textContent = "rudy didn't catch that. try again.";
      updateOrb('RUDY'); setExpression('eyeroll');
      return;
    }

    convoHistory.push(spoken);
    if (convoHistory.length > 6) convoHistory.shift();
    document.getElementById('micStatus').textContent = '';

    const special = checkSpecialTriggers(spoken);
    await withLoading('cooking something up...', 850);
    displayRoast(special || pickRoast(spoken));
  };

  recognition.onerror = (err) => {
    isListening = false;
    document.getElementById('micBtn').classList.remove('recording');
    if (err.error === 'not-allowed') {
      document.getElementById('micStatus').textContent = 'mic blocked. go to Settings > Safari > Microphone and allow.';
    } else if (err.error === 'no-speech') {
      document.getElementById('micStatus').textContent = 'rudy heard nothing. speak up fr.';
    } else if (err.error === 'aborted') {
      document.getElementById('micStatus').textContent = '';
    } else {
      document.getElementById('micStatus').textContent = 'error: ' + err.error + ' — try again.';
    }
    updateOrb('RUDY'); setExpression('annoyed');
  };

  try { recognition.start(); }
  catch(err) {
    isListening = false;
    document.getElementById('micBtn').classList.remove('recording');
    document.getElementById('micStatus').textContent = 'mic start failed. allow mic then retry.';
  }
}

function stopListening(e) {
  if (e) { e.preventDefault(); e.stopPropagation(); }
  if (!isListening || !recognition) return;
  try { recognition.stop(); } catch(err) {}
}

// Touch tracking for mobile
let touchX = 0, touchY = 0;
document.addEventListener('touchmove', (e) => {
  touchX = e.touches[0].clientX;
  touchY = e.touches[0].clientY;
  const orb = document.getElementById('orb');
  if (!orb) return;
  const rect = orb.getBoundingClientRect();
  const cx = rect.left + rect.width/2;
  const cy = rect.top + rect.height/2;
  const dx = (touchX - cx) / window.innerWidth;
  const dy = (touchY - cy) / window.innerHeight;
  const maxMove = 5;
  const px = 5 + dx * maxMove;
  const py = 5 + dy * maxMove;
  ['pupilL','pupilR'].forEach(id => {
    document.getElementById(id).style.left = px + 'px';
    document.getElementById(id).style.top = py + 'px';
  });
}, {passive:true});
